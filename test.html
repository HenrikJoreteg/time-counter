Count down (test 2)

Code:


var countDown = new TimeCounter({
        direction: 'down',
        startValue: '0:30'
    }),
    outputEl = document.getElementById("countDownOutput");

// log out all events
countDown.on('*', console.log.bind(console));

// set output value in output el
countDown.on('change', function (val) {
    outputEl.innerHTML = val;
});

countDown.start();

// alert when done
countDown.on('done', function () {
    alert('Count down done!');
});
